│       ├── 🏛️ class/            # WebSocket Classes
│       ├── 🛠️ utils/            # Utility Functions
│       └── 🏗️ server/           # WebSocket Server
│
├── 🗄️ database/                 # Database Layer
│   ├── 📊 schemas/              # Prisma Database Schemas
│   │   └── 🗂️ prisma/          # Prisma Configuration

│   │       └── �️ lib/          # Utility Functions
│   │
│   ├── �🔄 redux-toolkit/       # 📦 State Management + Persistence
│   │   ├── 🏪 store/           # Redux Store Configuration
│   │   ├── 🧩 slices/          # Redux Slices
│   │   ├── � thunks/          # Async Actions
│   │   └── 🪝 hooks/           # Custom Redux Hooks
│   │
│   ├── �📝 common/              # 🔄 Shared Zod Schemas & Types
│   │   ├── 📋 types/           # TypeScript Type Definitions
│   │   ├── 🔐 environment/     # Environment Variables
│   │   ├── 🎯 constants/       # Application Constants
│   │   └── 🎪 events/          # Event Schemas
│   │
│   ├── 🎨 tailwind/            # 🎨 Tailwind CSS Configuration

    ├── 🎯 eslint.config.js    # Global ESLint Rules
    ├── 📋 tsconfig.json       # TypeScript Root Config
